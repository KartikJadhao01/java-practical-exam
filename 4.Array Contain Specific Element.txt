public class SpecificEle {
public static void main(String[] args) {
	int arr[] = {10,20,30};
	
	int a = 20;
	
	for(int i = 0; i < arr.length; i++) {
		int c = 0;
		if(arr[i] == a) {
			c++;
		}
		if(c>0) {
			System.out.println(arr[i]);
		}
		
	}
	
	
	
}
}
