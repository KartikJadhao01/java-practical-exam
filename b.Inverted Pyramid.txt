=>public class InvertedPyramid {
public static void main(String[] args) {
	int n = 5;
	
	for(int i = n - 1; i >= 1; i--) {
		for(int s = 1; s <= n - i; s++) {
			System.out.print(" ");
		}
		for(int j = 1; j <= 2 * i - 1; j++) {
			if(i % 2 == 0) {
				System.out.print("1");
			}
			else {
				System.out.print("0");
			}
		}
		System.out.println();
	}
}
}
